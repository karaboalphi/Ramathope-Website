    // Handle admissions form submission (client-side only)
    function handleAdmission(e){
      e.preventDefault();
      const form = e.target;
      // Basic validation (already enforced by required attributes)
      const data = {
        fullname: form.fullname.value.trim(),
        guardian: form.guardian.value.trim(),
        email: form.email.value.trim(),
        phone: form.phone.value.trim(),
        grade: form.grade.value,
        message: form.message.value.trim(),
        reportName: form.report.files[0] ? form.report.files[0].name : 'No file uploaded'
      };

      // Show a friendly confirmation message (client-side)
      const result = document.getElementById('admission-result');
      result.style.display = 'block';
      result.innerHTML = '<h3>Application received (preview)</h3>'
        + '<p><strong>Applicant:</strong> ' + escapeHtml(data.fullname) + '</p>'
        + '<p><strong>Guardian:</strong> ' + escapeHtml(data.guardian) + ' -- ' + escapeHtml(data.email) + '</p>'
        + '<p><strong>Phone:</strong> ' + escapeHtml(data.phone) + '</p>'
        + '<p><strong>Grade:</strong> ' + escapeHtml(data.grade) + '</p>'
        + '<p><strong>Uploaded file:</strong> ' + escapeHtml(data.reportName) + '</p>'
        + '<p class="muted">This is a client-side confirmation only. To actually send the application to school email or database, integrate the form with a server-side script (PHP) or a form service. Ask me and I will provide the server code.</p>';
      result.scrollIntoView({behavior:'smooth',block:'center'});

      // keep form data local only (no send) -- return false
      return false;
    }

    // small helper to escape HTML
    function escapeHtml(s){
      return String(s).replace(/[&<>"']/g, function(m){ return {'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]; });
    }
